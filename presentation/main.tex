\section{Project Overview}
%------------------------------------------------------------
\begin{frame}
    \frametitle{Motivation: Why News Recommendation?}

    \begin{block}{The Problem}
        In the digital era, news platforms face severe \textbf{Information Overload}.
    \end{block}

    \vspace{0.2cm}
    \begin{block}{Why It Matters}
        \begin{itemize}
            \item \textbf{User Experience:} Navigating thousands of daily articles is overwhelming.
            \item \textbf{Retention:} Personalized content increases user engagement and platform stickiness.
            \item \textbf{Monetization:} Better engagement leads to higher ad revenue and subscriptions.
        \end{itemize}
    \end{block}

    \vspace{0.3cm}
    \begin{alertblock}{Overall Goal}
        Develop a state-of-the-art recommendation engine for \textbf{Vietnamese news}.
    \end{alertblock}
\end{frame}


\begin{frame}
    \frametitle{Problem Statement}

    \begin{block}{Research Question}
        How can we recommend \textbf{relevant news articles} to users effectively?
    \end{block}

    \vspace{0.3cm}
    \begin{block}{Key Challenges}
        \begin{enumerate}
            \item \textbf{Cold Start:} New users and articles have no interaction history.
            \item \textbf{Sparse Data:} Most users read very few articles.
            \item \textbf{Vietnamese Content Processing:} Limited resources and challenges in processing Vietnamese textual content.
            \item \textbf{Temporal Nature of News:} News articles are highly time-sensitive with a short lifespan.
        \end{enumerate}
    \end{block}
\end{frame}

\section{Motivation: Graph Expansion Strategy}
%------------------------------------------------------------

\begin{frame}
    \frametitle {Graph Structure}

    \begin{block}{Core Idea}
        To alleviate extreme sparsity, we \textbf{progressively enrich the interaction graph}.
        We define \textbf{three graph tiers (G1–G3)} for controlled experimentation.
    \end{block}

    \vspace{0.3cm}

    \begin{enumerate}
        \setlength\itemsep{0.9em}

        \item \textbf{Graph G1 — Bipartite Baseline}
              \begin{itemize}
                  \item \textbf{Nodes:} Users, Articles
                  \item \textbf{Edges:} \textit{Comment}
                  \item Represents a \textit{standard collaborative filtering setting}
              \end{itemize}

        \item \textbf{Graph G2 — Social Expansion (\textit{``The Increase''})}
              \begin{itemize}
                  \item \textbf{+ User–User Edges} via \textit{Reply} interactions
                  \item \textit{Hypothesis:} Users who converse are likely to share interests
              \end{itemize}

        \item \textbf{Graph G3 — Semantic Expansion}
              \begin{itemize}
                  \item \textbf{+ Category Nodes} connected to users they read
                  \item \textit{Hypothesis:} High-level semantic hubs reduce graph path length
              \end{itemize}

    \end{enumerate}
\end{frame}

\begin{frame}
    \frametitle{The Sparsity Challenge}
    \textbf{Problem:} Interactions are extremely sparse (99.93\%), making it hard for models to learn.

    \vspace{0.3cm}
    \begin{block}{Sparsity Calculation}
        \[ \text{Sparsity} = 1 - \frac{40,045 \text{ (Interactions)}}{15,693 \text{ (Users)} \times 3,645 \text{ (Articles)}} = \mathbf{99.93\%} \]
    \end{block}
\end{frame}

\begin{frame}
    \frametitle{Current SOTA Approaches}
    We benchmark against three main categories of modern Recommender Systems:
    \vspace{0.2cm}
    \begin{enumerate}
        \item \textbf{Graph Neural Networks (GNNs):}
              \begin{itemize}
                  \item \textbf{LightGCN \cite{he2020lightgcn} (2020):} Simplified GCN by removing non-linearities. Current industry standard for efficiency.
              \end{itemize}

              \vspace{0.2cm}
        \item \textbf{Contrastive Learning (CL):}
              \begin{itemize}
                  \item \textbf{SimGCL \cite{yu2022graph} (2022):} Augments graph structure with noise to learn robust representations. Handles sparsity better than pure GNNs.
                  \item \textbf{LightGCL \cite{cai2023lightgcl} (2023):} Efficient Global Context Learning via SVD for better structural alignment.
                  \item \textbf{XSimGCL \cite{yu2023xsimgcl} (2023):} Extreme Simple Graph Contrastive Learning using noise perturbation instead of graph augmentation.
              \end{itemize}

              \vspace{0.2cm}
        \item \textbf{Content-Based \& Baselines:}
              \begin{itemize}
                  \item \textbf{PhoBERT:} State-of-the-art BERT model for Vietnamese, capturing deep semantic text features.
                  \item \textbf{BGE-M3 \cite{xiao2023bgem3}:} Strong multilingual embedding model (dense/sparse) effective for global retrieval.
              \end{itemize}
    \end{enumerate}
\end{frame}

\begin{frame}
    \frametitle{State-of-the-Art Baselines: Contrastive Learning}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \begin{block}{1. LightGCL (Global Context)}
                \textbf{Problem:} Standard GNNs only capture local neighbors. \\
                \textbf{Solution:} Learn \textbf{Global Context} via SVD ($G = U_k \Sigma_k V_k^T$). \\
                \textbf{Why News?} Breaks "Info Cocoons" by recommending distinct but related topics.
            \end{block}
        \end{column}
        \begin{column}{0.48\textwidth}
            \begin{block}{2. XSimGCL (Extreme Simple)}
                \textbf{Problem:} Graph augmentation (drop edge) is slow. \\
                \textbf{Solution:} Add random \textbf{Noise} ($\Delta$) to each embeddings layers: $e' = e + \Delta$. \\
                \textbf{Benefit:} Fast (34s/epoch) \& Robust regularization.
            \end{block}
        \end{column}
    \end{columns}

    \vspace{0.2cm}
    \centering
    \small \textit{Both methods outperform traditional GCNs by improving representations.}
\end{frame}




\begin{frame}
    \frametitle{HGRec: Heterogeneous Graph Neural Network}
    \textbf{Core Idea:} Real-world graphs have multiple node/edge types (e.g., User-Reply-User vs. User-Comment-Article). Standard GNNs ignore this distinction \cite{shi2020hgrec}.

    \vspace{0.1cm}
    \begin{columns}
        \begin{column}{0.48\textwidth}
            \begin{block}{1. Meta-path Aggregation}
                Separately aggregates neighbors for each relation type:
                \[ h_u^{\Phi} = \text{AGG}\{h_v | v \in \mathcal{N}_u^{\Phi}\} \]
                \textit{Example:} Captures "Social" signals separately from "Interest" signals.
            \end{block}
        \end{column}

        \begin{column}{0.48\textwidth}
            \begin{block}{2. Semantic Fusion}
                Combines type-specific embeddings using Attention:
                \[ h_u^{Final} = \sum_{\Phi} \beta_{\Phi} \cdot h_u^{\Phi} \]
                \textit{Benefit:} Learns which relation is more important (e.g., Social > Interest for inactive users).
            \end{block}
        \end{column}
    \end{columns}

    \vspace{0.2cm}
    \textbf{Significance:} This mechanism forms the backbone of our \textbf{HGN} and \textbf{HCL} models.
\end{frame}

\begin{frame}
    \frametitle{Inspiration \& Theoretical Foundation}
    \textbf{Where did this idea come from?}
    Our architecture is an adaptation of the \textbf{HGRec} framework \cite{shi2020hgrec}, optimized with LightGCN \cite{he2020lightgcn} mechanics.

    \vspace{0.3cm}
    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.46\textwidth}
            \begin{block}{1. HGRec (Heterogeneous Graph Neural Network)}
                \textbf{Heterogeneous GNN for Recommendation}
                \begin{itemize}
                    \item \textbf{Key Idea:} Users and items have multiple types of relationships (e.g., Comment vs Reply).
                    \item \textbf{Mechanism:} Use relation-specific aggregation steps.
                \end{itemize}
            \end{block}
        \end{column}

        \begin{column}{0.46\textwidth}
            \begin{block}{2. LightGCN}
                \textbf{Simplified GCN for Recommendation}
                \begin{itemize}
                    \item \textbf{Key Idea:} Non-linearities (ReLU) are effectively redundant \\
                          for collaborative filtering.
                    \item \textbf{Mechanism:} Linear propagation + Layer averaging.
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}

    \vspace{0.2cm}
    \centering
    $\Rightarrow$ \textbf{HGN = HGRec's Multi-Relation logic + LightGCN's Efficiency}
\end{frame}


\begin{frame}
    \frametitle{Motivation: Why a Heterogeneous GNN?}
    \textbf{The Problem with Standard GNNs (LightGCN):}
    \begin{itemize}
        \item They treat the graph as \textbf{Homogeneous}: All nodes are just "IDs", all edges are just "links".
        \item \color{red} \textbf{Failure Case:} \color{black} A "Reply" (social trust) is treated identically to a "Comment" (explicit interest).
    \end{itemize}

    \vspace{0.3cm}
    \textbf{The VnExpress Reality:}
    \begin{itemize}
        \item \textbf{Multiple Node Types:} Users $\ne$ Items.
        \item \textbf{Multiple Edge Types:}
              \begin{itemize}
                  \item \texttt{User-Comment-Item}: Explicit Interest.
                  \item \texttt{User-Reply-User}: Social Influence / Trust.
              \end{itemize}
    \end{itemize}

    \vspace{0.2cm}
    \begin{block}{The HGN Solution}
        We need an architecture that aggregates \textbf{Social Signals} separately from \textbf{Content Signals} before fusing them.
    \end{block}
\end{frame}

\begin{frame}
    \frametitle{Evaluation Methodology \& Validity}
    Our evaluation follows strict scientific standards for Recommender Systems:

    \vspace{0.3cm}
    \begin{itemize}
        \item \textbf{Metric Choice:}
              \begin{itemize}
                  \item \textbf{Recall@10}: Measures system coverage.
                  \item \textbf{Hit Rate@10}: Measures if the relevant item is in the top-10.
                  \item \textbf{NDCG@10}: Rewards models that rank true items higher.
                  \item \textbf{MRR}: Standard for single-target news retrieval.
              \end{itemize}
        \item \textbf{Implicit Feedback Validation:}
              \begin{itemize}
                  \item These metrics are standard for \textbf{Binary (Implicit)} interactions.
                  \item NDCG compensates for varying test set sizes across users.
              \end{itemize}
        \item \textbf{Evaluation Strategy:}
              \begin{itemize}
                  \item \textbf{Full ranking (80/20):} Time user split. Train on 80\% of users, test on 20\% unseen users to measure zero-shot capability.
              \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Understanding Evaluation Protocols}

    \vspace{0.3cm}

              \begin{itemize}
                  \item \textbf{Setup:} Rank \textbf{all} candidate articles ($\approx$ 13,000+) for each test user.
                  \item \textbf{Challenge:} Extremely high noise; measures the model's ability to discriminate the relevant article from the entire corpus.
                  \item \textbf{Goal:} Real-world production performance (Cold-start \& Discovery).
              \end{itemize}
        \item \textbf{Loo100 Benchmark (Leave-One-Out):}
              \begin{itemize}
                  \item \textbf{Setup:} 1 relevant item + 99 random negative items for each test user.
                  \item \textbf{Challenge:} Localized selection; measures the model's sensitivity to distinguishing ground truth from standard noise.
                  \item \textbf{Goal:} Stability and fine-grained ranking precision (Sensitivity Analysis).
              \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Conclusion}
    \textbf{Key Achievements:}
    \begin{itemize}
        \item Built an end-to-end \textbf{Heterogeneous Graph Recommendation System} for Vietnamese News.
        \item Integrated \textbf{Social \& Temporal} signals, resolving the user cold-start and sparsity problem.
    \end{itemize}

    \vspace{0.4cm}
    \textbf{Limitations:}
    \begin{itemize}
        \item \textbf{Ranking Divergence:} Graph models struggle with \textbf{Global Retrieval} (Full Ranking) due to sparsity, but excel at \textbf{Personalized Refinement} (LOO).
        \item \textbf{Computational Cost:} Graph construction and neighborhood aggregation (GNN) are resource-intensive for large-scale graphs.
        \item \textbf{Offline Latency:} The current model is static; new articles require re-training or expensive incremental updates (no real-time adaptation).
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Future Directions}
    \textbf{Roadmap for Improvement:}
    \begin{itemize}
        \item \textbf{Dynamic Graph Learning (dyHGCN):} Capture evolving user interests over time instead of static snapshots.
        \item \textbf{More Meta Path:} Use more meta paths to capture different types of relationships between entities.
        \item \textbf{Self-Supervised Learning (SSL):} Apply contrastive learning to de-noise the graph and improve robustness against sparsity.
        \item \textbf{Knowledge Distillation:} Compress the large GNN model into a lightweight student model for faster inference.
    \end{itemize}
\end{frame}
