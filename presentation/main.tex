\section{Project Overview \& Methodology}
%------------------------------------------------------------
\begin{frame}
    \frametitle{Motivation: Why News Recommendation?}
    In the digital era, news platforms face \textbf{Information Overload}:

    \vspace{0.3cm}
    \begin{itemize}
        \item \textbf{User Experience:} Navigating thousands of daily articles is overwhelming.
        \item \textbf{Retention:} Personalized content increases user engagement and platform stickiness.
        \item \textbf{Monetization:} Better engagement leads to higher ad revenue and subscriptions.
    \end{itemize}

    \vspace{0.5cm}
    \textbf{Target:} Build a system that understands both \textit{Content semantics} and \textit{User behavior}.
\end{frame}

\begin{frame}
    \frametitle{Research Objectives}
    Our goal is to develop a state-of-the-art recommendation engine for Vietnamese news:

    \vspace{0.3cm}
    \begin{itemize}
        \item \textbf{Hybrid Logic:} Combine Collaborative Filtering (CF) with Content-Based (CB) techniques.
        \item \textbf{Graph Learning:} Leverage GNNs to model complex user-article interactions.
        \item \textbf{Multimodal Fusion:} Incorporate Social (replies) and Temporal (time-decay) signals.
        \item \textbf{Vietnamese NLP:} Optimize PhoBERT embeddings for news semantics.
    \end{itemize}
\end{frame}


\begin{frame}
    \frametitle{Problem Statement}
    \textbf{Challenge:} How to recommend relevant news articles to users?

    \vspace{0.5cm}
    \textbf{Key Challenges:}
    \begin{enumerate}
        \item \textbf{Cold Start:} New users/articles have no interaction history
        \item \textbf{Sparse Data:} Most users read very few articles (99.9\% sparsity)
        \item \textbf{Vietnamese Language:} Limited NLP resources for Vietnamese text
        \item \textbf{Dynamic Content:} News articles are time-sensitive
    \end{enumerate}

    \vspace{0.5cm}
    \textbf{Our Approach:} Graph-based Contrastive Learning with PhoBERT embeddings
\end{frame}

\section{State-of-the-Art (SOTA) Baselines}
%------------------------------------------------------------
\begin{frame}
    \frametitle{Current SOTA Approaches}
    We benchmark against three main categories of modern Recommender Systems:
    \vspace{0.2cm}
    \begin{enumerate}
        \item \textbf{Graph Neural Networks (GNNs):}
              \begin{itemize}
                  \item \textbf{NGCF (2019):} First to encode collaborative signal into embeddings via graph propagation. High computational cost.
                  \item \textbf{LightGCN (2020):} Simplified GCN by removing non-linearities. Current industry standard for efficiency.
              \end{itemize}

              \vspace{0.2cm}
        \item \textbf{Contrastive Learning (CL):}
              \begin{itemize}
                  \item \textbf{SimGCL (2022):} Augments graph structure with noise to learn robust representations. Handles sparsity better than pure GNNs.
              \end{itemize}

              \vspace{0.2cm}
        \item \textbf{Content-Based \& Hybrid:}
              \begin{itemize}
                  \item \textbf{PhoBERT:} State-of-the-art BERT model for Vietnamese, capturing deep semantic text features.
              \end{itemize}
    \end{enumerate}
\end{frame}

\begin{frame}
    \frametitle{Strategic Why: The Hybrid Synergy}
    Why combine \textbf{Content-Based} with \textbf{Graph Learning}?

    \vspace{0.4cm}
    \begin{itemize}
        \item \textbf{Content-Based (NLP):}
              \begin{itemize}
                  \item Solves the \textbf{Cold-Start} problem (recommends new articles with 0 interactions).
                  \item Captures semantic similarity (PhoBERT).
              \end{itemize}
        \item \textbf{Graph Learning (GNN):}
              \begin{itemize}
                  \item Captures \textbf{Collaborative Signals} (Users who like X also like Y).
                  \item Exploits \textbf{Social Signals} (Reply networks).
              \end{itemize}
    \end{itemize}

    \vspace{0.4cm}
    \begin{block}{Hybrid Outcome}
        Content-Based provides "Freshness" while Graph Learning provides "Depth" and "Discovery".
    \end{block}
\end{frame}

\begin{frame}
    \frametitle{Core Innovation: Social Fusion}
    \textbf{Idea:} Use the "Reply Network" to bridge gaps between users.

    \vspace{0.3cm}
    \begin{columns}
        \column{0.5\textwidth}
        \centering
        \begin{tikzpicture}[node distance=1.5cm, every node/.style={scale=0.7}]
            \node[circle, draw, fill=blue!20] (U1) {User A};
            \node[circle, draw, fill=blue!20, right of=U1, xshift=1cm] (U2) {User B};
            \node[rectangle, draw, fill=green!20, below of=U1] (A1) {Article 1};
            \node[rectangle, draw, fill=green!20, below of=U2] (A2) {Article 2};

            \draw[->, thick] (U1) -- (A1) node[midway, left] {Read};
            \draw[->, thick, red, dashed] (U1) -- (U2) node[midway, above] {Reply};
            \draw[->, thick] (U2) -- (A2) node[midway, right] {Read};

            \draw[->, blue, dotted, bend left] (U1) to node[midway, below] {Suggest?} (A2);
        \end{tikzpicture}

        \column{0.5\textwidth}
        \textbf{The Logic:}
        \begin{itemize}
            \item \textbf{Direct:} User A likes Article 1.
            \item \textbf{Social:} User A replies to User B $\rightarrow$ \textit{Shared Interest}.
            \item \textbf{Inference:} User A is likely to enjoy Article 2 (read by B).
        \end{itemize}
    \end{columns}

    \vspace{0.4cm}
    \begin{block}{Impact on Sparsity}
        By adding User-User edges, the GNN can propagate "interests" even if User A has only read 1 article.
    \end{block}
\end{frame}

\begin{frame}
    \frametitle{The Sparsity Challenge: Mathematical Proof}
    \textbf{Definition:} \\
    Sparsity measures the lack of interactions in the potential link space.

    \vspace{0.3cm}
    \begin{block}{The Formula}
        \[ \text{Sparsity} = 1 - \frac{|\mathcal{E}_{obs}|}{|\mathcal{V}_u| \times |\mathcal{V}_i|} \]
    \end{block}

    \vspace{0.2cm}
    \textbf{VnExpress Graph Calculation:}
    \begin{itemize}
        \item Interactions ($|\mathcal{E}_{obs}|$): \textbf{38,505}
        \item Active Users ($|\mathcal{V}_u|$): \textbf{13,117}
        \item Active Articles ($|\mathcal{V}_i|$): \textbf{3,645}
        \item Total Possible Edges: $13,117 \times 3,645 = \mathbf{47,811,465}$
    \end{itemize}

    \vspace{0.2cm}
    \centering
    $\Rightarrow \text{Sparsity} = 1 - \frac{38,505}{47,811,465} = \mathbf{99.919\%}$
\end{frame}

\section{Proposed Method: MA-HGN}
%------------------------------------------------------------

\begin{frame}
    \frametitle{Inspiration & Theoretical Foundation}
    \textbf{Where did this idea come from?}
    Our architecture is an adaptation of the \textbf{HGNR} framework, optimized with LightGCN mechanics.

    \vspace{0.3cm}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \begin{block}{1. HGNR (arXiv:2009.00799)}
                \textbf{Heterogeneous GNN for Recommendation}
                \begin{itemize}
                    \item \textbf{Key Idea:} Users and items have multiple types of relationships (e.g., Click vs Reply).
                    \item \textbf{Mechanism:} Use relation-specific aggregation steps.
                \end{itemize}
            \end{block}
        \end{column}

        \begin{column}{0.48\textwidth}
            \begin{block}{2. LightGCN (He et al., SIGIR'20)}
                \textbf{Simplified GCN for Recommendation}
                \begin{itemize}
                    \item \textbf{Key Idea:} Non-linearities (ReLU) are effectively redundant for collaborative filtering.
                    \item \textbf{Mechanism:} Linear propagation + Layer averaging.
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}

    \vspace{0.2cm}
    \centering
    $\Rightarrow$ \textbf{MA-HGN = HGNR's Multi-Relation logic + LightGCN's Efficiency}
\end{frame}

\begin{frame}
    \frametitle{Motivation: Why a Heterogeneous GNN?}
    \textbf{The Problem with Standard GNNs (LightGCN, NGCF):}
    \begin{itemize}
        \item They treat the graph as \textbf{Homogeneous}: All nodes are just "IDs", all edges are just "links".
        \item \color{red} \textbf{Failure Case:} \color{black} A "Reply" (social trust) is treated identically to a "Click" (casual interest).
    \end{itemize}

    \vspace{0.3cm}
    \textbf{The VnExpress Reality:}
    \begin{itemize}
        \item \textbf{Multiple Node Types:} Users $\ne$ Items.
        \item \textbf{Multiple Edge Types:}
              \begin{itemize}
                  \item \texttt{User-Click-Item}: Explicit Interest.
                  \item \texttt{User-Reply-User}: Social Influence / Trust.
              \end{itemize}
    \end{itemize}

    \vspace{0.2cm}
    \begin{block}{The MA-HGN Solution}
        We need an architecture that aggregates \textbf{Social Signals} separately from \textbf{Content Signals} before fusing them.
    \end{block}
\end{frame}

\begin{frame}
    \frametitle{MA-HGN: Multi-Aspect Heterogeneous GNN}
    \begin{columns}
        \begin{column}{0.55\textwidth}
            \centering
            % TikZ Diagram
            \begin{tikzpicture}[scale=0.7, transform shape]
                % Nodes
                \node[circle, fill=blue!20, draw=blue, thick, minimum size=1cm] (U) at (0,0) {$u_i$};
                \node[circle, fill=green!20, draw=green!60!black, thick] (A) at (-2, 2) {$a_j$};
                \node[circle, fill=blue!10, draw=blue!50, dashed] (U2) at (2, 2) {$u_k$};

                % Aggregators
                \node[draw, rectangle, fill=orange!20, rounded corners] (GAT) at (-2, -1.8) {GAT (Interest)};
                \node[draw, rectangle, fill=purple!20, rounded corners] (SAGE) at (2, -1.8) {SAGE (Social)};
                \node[draw, rectangle, fill=red!20, rounded corners, minimum width=3cm] (FUSE) at (0, -3.5) {\textbf{Attention Fusion}};

                % Edges
                \draw[->, thick] (A) -- (U) node[midway, left, font=\footnotesize] {Behavior};
                \draw[->, thick, dashed] (U2) -- (U) node[midway, right, font=\footnotesize] {Social};

                % Flows
                \draw[->, orange, line width=1.5pt] (U) to[bend left] (GAT);
                \draw[->, purple, line width=1.5pt] (U) to[bend right] (SAGE);

                \node[below=0.2cm of FUSE] (OUT) {$h_u^*$ (Final)};
                \draw[->, thick] (FUSE) -- (OUT);
            \end{tikzpicture}
        \end{column}

        \begin{column}{0.45\textwidth}
            \textbf{Core Innovation: Dual-path Propagation}
            \vspace{0.2cm}
            \begin{itemize}
                \item \textbf{Path 1 (Interest):} history ($\mathcal{G}_{u-a}$) via \textbf{GATv2}.
                \item \textbf{Path 2 (Social):} friends ($\mathcal{G}_{u-u}$) via \textbf{GraphSAGE}.
                \item \textbf{Adaptive Fusion:} balances signal importance:
                      \[ h_u = \alpha \cdot h_{int} + (1-\alpha) \cdot h_{soc} \]
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}
    \frametitle{2. Model Training Pipeline}
    \textbf{Orchestration Script:} \texttt{train\_cf\_models.py}

    \vspace{0.3cm}
    \begin{block}{Training Loop (Epoch $t$)}
        \begin{enumerate}
            \item \textbf{Graph Encoder:} Pass graph $\mathcal{G}$ through $L$ layers of LightGCN/SimGCL.
            \item \textbf{Vector Retrieval:} Get User $E_u$ and Item $E_i$ embeddings.
            \item \textbf{Negative Sampling:} Sample hard negatives (items user didn't interact with).
            \item \textbf{Loss Computation:}
                  \[ \mathcal{L}_{total} = \mathcal{L}_{BPR}(pos, neg) + \lambda \cdot \mathcal{L}_{CL}(View_1, View_2) + \gamma \cdot \|\Theta\|_2 \]
            \item \textbf{Backpropagation:} Update weights via Adam Optimizer ($lr=0.001$).
        \end{enumerate}
    \end{block}
\end{frame}

\begin{frame}
    \frametitle{3. Deployment Pipeline (Streamlit)}
    \textbf{Real-time Inference Engine:}

    \vspace{0.3cm}
    \begin{columns}
        \column{0.5\textwidth}
        \begin{block}{Hybrid Scoring Logic}
            For User $u$ and Candidate Item $i$:
            \begin{enumerate}
                \item \textbf{CF Score:} Dot product $s_{cf} = e_u^T e_i$ (from SimGCL).
                \item \textbf{CB Score:} Cosine similarity $s_{cb}$ between User Profile vector and Item Vector (VN-SBERT).
                \item \textbf{Fusion:}
                      \[ S_{final} = \beta \cdot \text{norm}(s_{cf}) + (1-\beta) \cdot s_{cb} \]
            \end{enumerate}
        \end{block}

        \column{0.5\textwidth}
        \textbf{Features:}
        \begin{itemize}
            \item \textbf{Response Time:} $< 50ms$ (Pre-computed indices).
            \item \textbf{Cold-Start Fallback:} If user is new ($s_{cf}=0$), fallback to Trending or Content-Based.
            \item \textbf{Interactive UI:} Adjust $\beta$ weight dynamically.
        \end{itemize}
    \end{columns}
\end{frame}

    \begin{itemize}
        \item \textbf{Metric Choice:}
              \begin{itemize}
                  \item \textit{Recall@10}: Measures system coverage.
                  \item \textit{NDCG@10}: Rewards models that rank true items higher.

        \item \textbf{Split Strategy:} Time-aware 80/10/10 split to prevent leakage.

\begin{frame}
    \frametitle{Investigation: The "Projection Trap"}
    \textbf{Observed Paradox:} Standalone PhoBERT performed 10x worse than expected.

    \vspace{0.3cm}
    \begin{columns}
        \begin{column}{0.48\textwidth}
            \begin{block}{The Problem}
                \begin{itemize}
                    \item \textbf{Random Projection:} PhoBERT embeddings (768-dim) were compressed via untrained random linear layers.
                    \item \textbf{Noise Injection:} High-quality semantics were destroyed by "random noise" before retrieval.
                \end{itemize}
            \end{block}
        \end{column}

        \begin{column}{0.48\textwidth}
            \begin{block}{The Solution (Rescued)}
                \begin{itemize}
                    \item \textbf{Raw Embedding Retrieval:} Skipping untrained layers and using direct cosine similarity.
                    \item \textbf{Result:} Recall@10 jumped from \textbf{0.3\%} to \textbf{3.5\%} (\textbf{+1066\%} improvement).
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}
    \frametitle{Ablation: Embedding Initialization Impact}
    \textbf{Hypothesis:} Initializing GNN with PhoBERT should beat Random Init.
    \textbf{Result:} \textbf{Random Init (12.40\%) > PhoBERT Init (4.96\%)}.

    \vspace{0.3cm}
    \begin{itemize}
        \item \textbf{Semantic vs. Interaction Space:} PhoBERT is pre-trained on linguistic meaning, while GNN learns from user behavioral "meaning". These spaces are often misaligned.
        \item \textbf{Geometric Bias:} Initializing in a fixed semantic space "anchors" nodes, making it harder for the GNN to cluster users based on collaborative signals alone.
        \item \textbf{Discovery Power:} Randomly initialized embeddings have the freedom to map latent interests that text alone cannot describe.
    \end{itemize}

    \begin{block}{Verdict}
        Behavioral signals (who-reads-what) are significantly more informative than semantic signals for Vietnamese news.
    \end{block}
\end{frame}

    \frametitle{Discussion: TF-IDF vs PhoBERT (Final Comparison)}

    \begin{columns}
        \begin{column}{0.48\textwidth}

                \begin{itemize}
                    \item \textbf{Precision:} Perfect match for Vietnamese named entities (Tây Nguyên, EVN).

            \end{block}

\end{frame}
