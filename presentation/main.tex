\section{Project Overview \& Methodology}
%------------------------------------------------------------
\begin{frame}
    \frametitle{Motivation: Why News Recommendation?}
    In the digital era, news platforms face \textbf{Information Overload}:

    \vspace{0.3cm}
    \begin{itemize}
        \item \textbf{User Experience:} Navigating thousands of daily articles is overwhelming.
        \item \textbf{Retention:} Personalized content increases user engagement and platform stickiness.
        \item \textbf{Monetization:} Better engagement leads to higher ad revenue and subscriptions.
    \end{itemize}

    \vspace{0.5cm}
    \textbf{Target:} Build a system that understands both \textit{Content semantics} and \textit{User behavior}.
\end{frame}

\begin{frame}
    \frametitle{Research Objectives}
    Our goal is to develop a state-of-the-art recommendation engine for Vietnamese news:

    \vspace{0.3cm}
    \begin{itemize}
        \item \textbf{Hybrid Logic:} Combine Collaborative Filtering (CF) with Content-Based (CB) techniques.
        \item \textbf{Graph Learning:} Leverage GNNs to model complex user-article interactions.
        \item \textbf{Multimodal Fusion:} Incorporate Social (replies) signals.
        \item \textbf{Vietnamese NLP:} Optimize PhoBERT embeddings for news semantics.
    \end{itemize}
\end{frame}


\begin{frame}
    \frametitle{Problem Statement}
    \textbf{Challenge:} How to recommend relevant news articles to users?

    \vspace{0.5cm}
    \textbf{Key Challenges:}
    \begin{enumerate}
        \item \textbf{Cold Start:} New users/articles have no interaction history
        \item \textbf{Sparse Data:} Most users read very few articles (99.9\% sparsity)
        \item \textbf{Vietnamese Language:} Limited NLP resources for Vietnamese text
        \item \textbf{Dynamic Content:} News articles are time-sensitive
    \end{enumerate}

    \vspace{0.5cm}
    \textbf{Our Approach:} Graph-based Contrastive Learning with PhoBERT embeddings
\end{frame}

              \begin{itemize}
                  \item \textbf{Caching:} Avoids redundant HTTP requests for static content.
                  \item \textbf{Checkpointing:} Atomic writes to CSV allow resuming from the last article in case of failure.
                  \item \textbf{Headless Execution:} Minimizes resource overhead in production environments.
              \end{itemize}
        \item \textbf{Tool Stack:}
              \begin{itemize}
                  \item \textbf{Requests:} High-speed metadata extraction.
                  \item \textbf{Selenium:} Handling complex JS (Load More buttons).

              \end{itemize}
    \end{itemize}

    \frametitle{Current SOTA Approaches}
    We benchmark against three main categories of modern Recommender Systems:
    \vspace{0.2cm}
    \begin{enumerate}
        \item \textbf{Graph Neural Networks (GNNs):}
              \begin{itemize}
                  \item \textbf{LightGCL (2023):} Efficient Global Context Learning via SVD for better structural alignment.
                  \item \textbf{XSimGCL (2023):} Extreme Simple Graph Contrastive Learning using noise perturbation instead of graph augmentation.
                  \item \textbf{LightGCN (2020):} Simplified GCN by removing non-linearities. Current industry standard for efficiency.
              \end{itemize}

              \vspace{0.2cm}
        \item \textbf{Contrastive Learning (CL):}
              \begin{itemize}
                  \item \textbf{SimGCL (2022):} Augments graph structure with noise to learn robust representations. Handles sparsity better than pure GNNs.
              \end{itemize}

              \vspace{0.2cm}
        \item \textbf{Content-Based \& Hybrid:}
              \begin{itemize}
                  \item \textbf{PhoBERT:} State-of-the-art BERT model for Vietnamese, capturing deep semantic text features.
              \end{itemize}
    \end{enumerate}
\end{frame}

\section{Motivation: Graph Expansion Strategy}
%------------------------------------------------------------

\begin{frame}
    \frametitle{The Sparsity Challenge}
    \textbf{Problem:} Interactions are extremely sparse (99.919\%), making it hard for models to learn.

    \vspace{0.3cm}
    \begin{block}{Sparsity Calculation}
        \[ \text{Sparsity} = 1 - \frac{38,505 \text{ (Interactions)}}{13,117 \text{ (Users)} \times 3,645 \text{ (Articles)}} = \mathbf{99.919\%} \]
    \end{block}
\end{frame}

\begin{frame}
    \frametitle{Solution: Expanding the Graph Structure}
    We progressively enrich the graph to alleviate sparsity. We define \textbf{3 Graph Tiers} for experimentation:

    \vspace{0.3cm}
    \begin{itemize}
        \setlength\itemsep{1em}
        \item \textbf{Graph G1 (Bipartite Baseline):}
              \begin{itemize}
                  \item Nodes: Users, Articles.
                  \item Edges: \textit{Read, Comment, Share}.
                  \item \textit{Standard CF setting.}
              \end{itemize}

        \item \textbf{Graph G2 (Social Expansion - "The Increase"):}
              \begin{itemize}
                  \item \textbf{+ User-User Edges:} Created via "Reply" interactions.
                  \item \textit{Hypothesis: Users who converse share interests.}
              \end{itemize}

        \item \textbf{Graph G3 (Semantic Expansion):}
              \begin{itemize}
                  \item \textbf{+ Category Nodes:} Users connect to Categories they read.
                  \item \textit{Hypothesis: High-level hubs reduce path length.}
              \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Core Innovation: Social Fusion (Graph G2)}
    \textbf{Idea:} Use the "Reply Network" to bridge gaps between users.

    \vspace{0.3cm}
    \begin{columns}
        \column{0.5\textwidth}
        \centering
        \begin{tikzpicture}[node distance=1.5cm, every node/.style={scale=0.7}]
            \node[circle, draw, fill=blue!20] (U1) {User A};
            \node[circle, draw, fill=blue!20, right of=U1, xshift=1cm] (U2) {User B};
            \node[rectangle, draw, fill=green!20, below of=U1] (A1) {Article 1};
            \node[rectangle, draw, fill=green!20, below of=U2] (A2) {Article 2};

            \draw[->, thick] (U1) -- (A1) node[midway, left] {Read};
            \draw[->, thick, red, dashed] (U1) -- (U2) node[midway, above] {Reply};
            \draw[->, thick] (U2) -- (A2) node[midway, right] {Read};

            \draw[->, blue, dotted, bend left] (U1) to node[midway, below] {Suggest?} (A2);
        \end{tikzpicture}

        \column{0.5\textwidth}
        \textbf{The Logic:}
        \begin{itemize}
            \item \textbf{Direct:} User A likes Article 1.
            \item \textbf{Social:} User A replies to User B $\rightarrow$ \textit{Shared Interest}.
            \item \textbf{Inference:} User A is likely to enjoy Article 2 (read by B).
        \end{itemize}
    \end{columns}

    \vspace{0.4cm}
    \begin{block}{Impact on Sparsity}
        By adding User-User edges (G2), the GNN can propagate "interests" even if User A has only read 1 article.
    \end{block}
\end{frame}

\section{Proposed Method: HGN}
%------------------------------------------------------------

\begin{frame}
    \frametitle{Inspiration \& Theoretical Foundation}
    \textbf{Where did this idea come from?}
    Our architecture is an adaptation of the \textbf{HGNR} framework, optimized with LightGCN mechanics.

    \vspace{0.3cm}
    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.46\textwidth}
            \begin{block}{1. HGNR (arXiv:\allowbreak 2009.00799)}
                \textbf{Heterogeneous GNN for Recommendation}
                \begin{itemize}
                    \item \textbf{Key Idea:} Users and items have multiple types of relationships (e.g., Comment vs Reply).
                    \item \textbf{Mechanism:} Use relation-specific aggregation steps.
                \end{itemize}
            \end{block}
        \end{column}

        \begin{column}{0.46\textwidth}
            \begin{block}{2. LightGCN (He et al., SIGIR'20)}
                \textbf{Simplified GCN for Recommendation}
                \begin{itemize}
                    \item \textbf{Key Idea:} Non-linearities (ReLU) are effectively redundant \\
                          for collaborative filtering.
                    \item \textbf{Mechanism:} Linear propagation + Layer averaging.
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}

    \vspace{0.2cm}
    \centering
    $\Rightarrow$ \textbf{HGN = HGNR's Multi-Relation logic + LightGCN's Efficiency}
\end{frame}


\begin{frame}
    \frametitle{Motivation: Why a Heterogeneous GNN?}
    \textbf{The Problem with Standard GNNs (LightGCN):}
    \begin{itemize}
        \item They treat the graph as \textbf{Homogeneous}: All nodes are just "IDs", all edges are just "links".
        \item \color{red} \textbf{Failure Case:} \color{black} A "Reply" (social trust) is treated identically to a "Comment" (explicit interest).
    \end{itemize}

    \vspace{0.3cm}
    \textbf{The VnExpress Reality:}
    \begin{itemize}
        \item \textbf{Multiple Node Types:} Users $\ne$ Items.
        \item \textbf{Multiple Edge Types:}
              \begin{itemize}
                  \item \texttt{User-Comment-Item}: Explicit Interest.
                  \item \texttt{User-Reply-User}: Social Influence / Trust.
              \end{itemize}
    \end{itemize}

    \vspace{0.2cm}
    \begin{block}{The HGN Solution}
        We need an architecture that aggregates \textbf{Social Signals} separately from \textbf{Content Signals} before fusing them.
    \end{block}
\end{frame}

\begin{frame}
    \frametitle{HGN: Multi-Aspect Heterogeneous GNN}
    \begin{columns}
        \begin{column}{0.55\textwidth}
            \centering
            % TikZ Diagram
            \begin{tikzpicture}[scale=0.7, transform shape]
                % Nodes
                \node[circle, fill=blue!20, draw=blue, thick, minimum size=1cm] (U) at (0,0) {$u_i$};
                \node[circle, fill=green!20, draw=green!60!black, thick] (A) at (-2, 2) {$a_j$};
                \node[circle, fill=blue!10, draw=blue!50, dashed] (U2) at (2, 2) {$u_k$};

                % Aggregators
                \node[draw, rectangle, fill=orange!20, rounded corners] (GAT) at (-2, -1.8) {GAT (Interest)};
                \node[draw, rectangle, fill=purple!20, rounded corners] (SAGE) at (2, -1.8) {SAGE (Social)};
                \node[draw, rectangle, fill=red!20, rounded corners, minimum width=3cm] (FUSE) at (0, -3.5) {\textbf{Attention Fusion}};

                % Edges
                \draw[->, thick] (A) -- (U) node[midway, left, font=\footnotesize] {Behavior};
                \draw[->, thick, dashed] (U2) -- (U) node[midway, right, font=\footnotesize] {Social};

                % Flows

                \draw[->, purple, line width=1.5pt] (U) to[bend right] (SAGE);

                \draw[->, purple, line width=1.5pt] (SAGE) -- (FUSE);

                \node[below=0.2cm of FUSE] (OUT) {$h_u^*$ (Final)};
                \draw[->, thick] (FUSE) -- (OUT);
            \end{tikzpicture}
        \end{column}

        \begin{column}{0.45\textwidth}
            \textbf{Core Innovation: Dual-path Propagation}
            \vspace{0.2cm}
            \begin{itemize}
                \item \textbf{Path 1 (Interest):} history ($\mathcal{G}_{u-a}$) via \textbf{GATv2}.
                \item \textbf{Path 2 (Social):} friends ($\mathcal{G}_{u-u}$) via \textbf{GraphSAGE}.
                \item \textbf{Adaptive Fusion:} balances signal importance:
                      \[ h_u = \alpha \cdot h_{int} + (1-\alpha) \cdot h_{soc} \]
            \end{itemize}
        \end{column}
    \end{columns}

\begin{frame}
    \frametitle{Evaluation Methodology \& Validity}
    Our evaluation follows strict scientific standards for Recommender Systems:

    \vspace{0.3cm}
    \begin{itemize}
        \item \textbf{Metric Choice:}
              \begin{itemize}
                  \item \textit{Recall@10}: Measures system coverage.
                  \item \textit{NDCG@10}: Rewards models that rank true items higher.
                  \item \textbf{MRR}: Standard for single-target news retrieval.
              \end{itemize}
        \item \textbf{Dual Protocol Strategy:}
              \begin{itemize}
                  \item \textbf{Mainstream (LOO):} Temporal Leave-One-Out. Test on the \textit{latest} interaction to prevent future leakage.
                  \item \textbf{Cold-Start (80/20):} Random user split. Train on 80\% of users, test on 20\% unseen users to measure zero-shot capability.
              \end{itemize}
    \end{itemize}
\end{frame}
